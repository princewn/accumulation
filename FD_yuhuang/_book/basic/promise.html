//嵌套的使用
function masterRight(choise) {
    var defer = $.Deferred();
    (function() {
        var defer2 = $.Deferred();
        if (choise) {
            defer2.resolve(isMaster);
        } else {
            defer2.reject(isMaster);
        }
        return defer2.promise();
    })();
    return defer.promise();
}
